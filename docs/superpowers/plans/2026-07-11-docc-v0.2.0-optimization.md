```js
/**
 * 锚点边界匹配:行以 anchor 开头,且其后紧跟行尾或空白——
 * 防止"## 状态"被"## 状态更新记录"这类更长标题字面顶替(假阳性)。
 * 返回命中行数;读文件失败返回 0,不抛。
 */
export function countAnchorMatches(cwd, docFile, anchor) {
  try {
    const text = fs.readFileSync(path.join(cwd, docFile), "utf-8");
    return text.split(/\r?\n/).filter((line) => {
      if (!line.startsWith(anchor)) return false;
      const rest = line.slice(anchor.length);
      return rest === "" || /^\s/.test(rest);
    }).length;
  } catch {
    return 0;
  }
}

export function anchorExists(cwd, docFile, anchor) {
  return countAnchorMatches(cwd, docFile, anchor) > 0;
}
```

```js
/** 生成物/内部目录默认排除(目录前缀,恒生效);config.exclude 可追加自定义前缀 */
export const DEFAULT_EXCLUDES = [".understand-anything/", ".claude/"];

/** 默认排除 ∪ config.exclude 合法成员;归一化:trim、去 ./ 前缀、补尾 /;坏输入静默忽略 */
export function getExcludes(docMap) {
  const extra = Array.isArray(docMap?.config?.exclude)
    ? docMap.config.exclude
        .filter((p) => typeof p === "string" && p.trim())
        .map((p) => {
          let s = p.trim();
          if (s.startsWith("./")) s = s.slice(2);
          if (!s) return null;
          return s.endsWith("/") ? s : `${s}/`;
        })
        .filter(Boolean)
    : [];
  return [...DEFAULT_EXCLUDES, ...extra];
}

export function isExcluded(relPosix, excludes) {
  return excludes.some((p) => relPosix.startsWith(p));
}
```

```js
#!/usr/bin/env node
/**
 * 事中·契约门(PreToolUse, matcher: Write|Edit|MultiEdit)。
 * 仅拦 critical 锚点关联文件的首次编辑:deny 一次、机写声明 stub 草稿,
 * 要求在消息中声明文档影响并把草稿中 pending 占位替换为同一声明,
 * 原样重试即放行;同会话同文件不再拦。
 * 状态写入失败必须降级放行——宁可少拦一次,绝不造成重试死锁。
 */
import path from "node:path";
import {
  loadDocMap,
  matchEntries,
  repoRelative,
  loadState,
  saveState,
  declarationsPath,
  appendDeclarationStub,
  readStdinJson,
  toPosix,
  warn,
} from "./_shared.mjs";

try {
  const payload = readStdinJson();
  const cwd = payload?.cwd || process.cwd();
  const filePath = payload?.tool_input?.file_path;
  if (!filePath) process.exit(0);

  const docMap = loadDocMap(cwd);
  if (!docMap) process.exit(0);

  const rel = repoRelative(filePath, cwd);
  if (!rel) process.exit(0);

  const criticalAnchors = matchEntries(docMap, rel).flatMap((e) =>
    e.docs.filter((d) => d.critical).map((d) => `${d.file} ${d.anchor ?? ""}`.trim()),
  );
  if (!criticalAnchors.length) process.exit(0);

  const state = loadState(cwd, payload?.session_id);
  if (state.gated.includes(rel)) process.exit(0); // 已陈述过:放行
  state.gated.push(rel);
  if (!saveState(cwd, payload?.session_id, state)) {
    warn("契约门状态不可写,本次降级放行");
    process.exit(0);
  }

  const stubWritten = appendDeclarationStub(cwd, payload?.session_id, rel, criticalAnchors);
  const draftRel = toPosix(path.relative(cwd, declarationsPath(cwd, payload?.session_id)));
  const draftStep = stubWritten
    ? `并把声明草稿 ${draftRel} 中本文件条目的 <!-- pending --> 替换为同一声明(postflight 对账入台账),`
    : "";
  process.stderr.write(
    `[doc-companion 契约门] ${rel} 关联契约级文档锚点:${criticalAnchors.join("、")}。\n` +
      "请在消息文本中声明本次改动对上述锚点的影响(有 → 将同步哪一节;无 → 理由)," +
      draftStep +
      "然后原样重试本次编辑即放行;同会话本文件不再拦。",
  );
  process.exit(2);
} catch (e) {
  warn(`gate fail-open: ${e?.message}`);
  process.exit(0);
}
```

````markdown
## 步骤

1. **摸底**:读仓库目录结构(一层)与现有文档(README、docs/ 下的手册/契约/设计文档),找出"记载了代码行为"的文档及其章节结构。
2. **对话确认**(逐项问用户,一次一问):
   - 哪些文档是行为真相源?(默认候选:README 状态节、docs/ 下的契约/附录类文档)
   - 哪些章节属于"契约冻结级"(改动必须声明影响)→ 这些锚点标 `critical: true`
   - diff 基线分支(默认 `main`)
   - 是否有需要排除的生成物/产物目录(写入 `config.exclude`;`.understand-anything/` 与 `.claude/` 已默认排除)
3. **生成 `docs/DOC-MAP.json`**,结构:

```json
{
  "version": 1,
  "config": {
    "ledgerDir": "docs/LEDGER",
    "diffBase": "main",
    "exclude": ["dist/"],
    "history": true
  },
  "entries": [
    {
      "pattern": "src/api/**",
      "docs": [
        { "file": "docs/appendix.md", "anchor": "### 2.6", "note": "API 契约", "critical": true },
        { "file": "README.md", "anchor": "## 状态" }
      ]
    }
  ]
}
```

规则:
- `pattern` 优先目录级 glob(新增文件自动纳管);glob 只支持 `**`/`*`/`?`,**不支持** `!` 排除、`{a,b}` 展开、`[abc]` 字符类——目录级排除用 `config.exclude`(目录前缀数组,非 glob),模式级排除需拆分正向条目。
- `anchor` 是目标文档中标题行的前缀,匹配要求**前缀之后是行尾或空白**(`"## 状态"` 不会命中 `"## 状态更新记录"`);同一锚点在文档中命中 ≥2 次会被 preflight 报"锚点歧义"。
- `config.history` 缺省为 true:每次盖章向 `docs/DOC-MAP.history.jsonl` 追加一行漂移计数(观测数据,建议随仓库提交);设 `false` 关闭。
- DOC-MAP 自身与 hashes 文件也要登记为一个 critical 条目(自守护);hashes/history 文件不参与源侧漂移追踪(脚本恒排除,防自引用)。

5. **建哈希基线**:`node "${CLAUDE_PLUGIN_ROOT}/scripts/doc-preflight.mjs" --cwd . --stamp`,把生成的 `docs/DOC-MAP.json`、`docs/DOC-MAP.hashes.json` 与 `docs/DOC-MAP.history.jsonl`(默认开启时)提交到项目仓库。
6. **输出接线建议**(供用户贴进项目 AGENTS.md/CLAUDE.md):phase/迭代开工首任务 = `/docc:preflight`,收尾任务 = `/docc:postflight`(在代码审查之前跑,使台账可供审查引用)。
````

````markdown
# doc-companion

让 AI 写代码时文档始终随行的 Claude Code 插件:**事前**校对文档与真实源码的漂移(含"文档被人为直接修改"的检测)、**事中**提醒与契约门(声明落盘可审计)、**事后**变更台账与索引校正,全部围绕一张代码↔文档锚点映射表(DOC-MAP)工作。

## 安装

```
/plugin marketplace add Aqrk-Dev/doc-companion
/plugin install docc@doc-companion
```

## 接入一个项目(每仓库一次)

```
/docc:init
```

init 会对话式生成 `docs/DOC-MAP.json`(映射 + `critical` 契约标记 + `config`),建立 `docs/DOC-MAP.hashes.json` 哈希基线(`git hash-object`,双侧:源码+文档),并输出工作流接线建议。**未 init 的项目装了插件零打扰**(hooks 检测不到 DOC-MAP 即静默退出)。

## 四环节

| 环节 | 入口 | 行为 |
|---|---|---|
| 事前 | `/docc:preflight` | 哈希比对锁定漂移候选(源码变过 / 文档被直接改过),只对候选做语义对拍,漂移归零再开工 |
| 事中 | hooks 自动 | 改到映射文件→提醒一次(每文件每会话);改 `critical` 文件→拦一次要求声明文档影响,同时机写声明 stub 到 `.claude/.cache/doc-companion-<session>.declarations.md` 供 postflight 对账,重试放行;每轮收尾→未处置清单一行提醒(不阻塞) |
| 事后 | `/docc:postflight` | 按 diff 更新受影响锚点 → 合并契约门声明草稿 → 五字段台账(+声明对账节,verified-by 标记)→ 索引校正 → 重新盖章 |
| 索引 | preflight/postflight 内建 | 锚点悬空、锚点歧义、pattern 失配、孤儿清理 |

## 设计红线

- **fail-open**:任何脚本异常、状态损坏、运行时缺失 = 放行 + stderr 告警,绝不阻塞开发;契约门在状态不可写时降级放行(防重试死锁);声明 stub/历史埋点写失败只产生软告警。
- **同会话同文件只拦一次**:迭代式"编辑→测试→再编辑"不受契约门干扰(测试回归钉钉死)。
- **运行时分层**:核心链路(preflight/postflight skill)只依赖 git;hooks 需要 node(≥18,零 npm 依赖),机器无 node 时经 `command -v` 探测静默哑火,skill 按降级路径手工执行同一清单。
- **引擎零项目耦合**:本仓库代码不含任何具体项目信息;项目特定内容全部在各仓库自己的 DOC-MAP 数据里。

## DOC-MAP 结构

```json
{
  "version": 1,
  "config": {
    "ledgerDir": "docs/LEDGER",
    "diffBase": "main",
    "exclude": ["dist/"],
    "history": true
  },
  "entries": [
    {
      "pattern": "src/api/**",
      "docs": [
        { "file": "docs/appendix.md", "anchor": "### 2.6", "note": "API 契约", "critical": true },
        { "file": "README.md", "anchor": "## 状态" }
      ]
    }
  ]
}
```

- `pattern`:极简 glob(`**` 跨段 / `*` 段内 / `?` 单字符)。
- `anchor`:目标文档标题行前缀,**边界匹配**——前缀之后必须是行尾或空白(`"## 状态"` 不命中 `"## 状态更新记录"`);同一锚点命中 ≥2 处报"锚点歧义"。
- `critical`:事中契约门开关。
- `config.exclude`:目录前缀排除数组(叠加在默认排除 `.understand-anything/`、`.claude/` 之上);`docs/DOC-MAP.hashes.json` 与 `docs/DOC-MAP.history.jsonl` 恒排除(防盖章自引用漂移)。
- `config.history`:缺省 true——盖章时向 `docs/DOC-MAP.history.jsonl` 追加一行漂移计数(`ts/driftCandidates/docDrift/mapIssues/warnings/stamped`),为"格式化误报是否真实出现"等长期问题提供观测数据,建议随仓库提交;`false` 关闭。

## 报告字段

`scripts/doc-preflight.mjs` 输出 JSON:`ok`(mapIssues 为空)、`driftCandidates`、`docDrift`、`mapIssues`(含锚点歧义)、`warnings`(软告警:台账缺 `verified-by` 标记、历史埋点写失败;不影响 `ok`)、`stamped`。

## 已知限制

1. **glob 无排除语法**:不支持 `!`、`{a,b}`、`[abc]`。目录级排除用 `config.exclude`(前缀,非 glob);`**/*_test.go` 这类模式级排除做不到,需把正向条目拆细。
2. **多 agent/worktree 并发不合并声明**:多个并行会话各自触发契约门、各写各的声明草稿,docc 不自动对账矛盾声明——postflight 合并草稿时需人工裁决。
3. **仅 Claude Code 会话内生效**:三档 hook 挂在 Claude Code 事件上;人类直接 commit 或未装插件的协作者提交不经任何门禁,靠下一次 preflight 被动发现(CI 门禁未内置)。
````
